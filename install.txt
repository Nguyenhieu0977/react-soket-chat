0. Nginx: nginx-setup.conf

upstream api {
    server backend:8000;
}

server {
  listen 8443;

  location / {
    root /var/www/client;
  }

  location /api/ {
    proxy_pass http://;
    proxy_set_header Host $http_host;
  }

}

1. Dockerfile cntttqk7
# syntax=docker/dockerfile:1
FROM python:3
ENV PYTHONUNBUFFERED=1
WORKDIR /cntttqk7
COPY requirements.txt /cntttqk7/
RUN pip install -r requirements.txt
COPY . /cntttqk7/

2. Dockerfile Client
FROM node:15.13-alpine
WORKDIR /client
COPY package.json ./
COPY package-lock.json ./
COPY . .
RUN npm install
COPY . .
RUN npm run build

3. docker-compose.yml
version: '3'

services:
  backend:
    build:
      context: ./cntttqk7
    command: python manage.py runserver 0.0.0.0:8000
    ports:
      - "8000:8000"
  frontend:
    build:
      context: ./client
    volumes:
      - react_build:/client/build
  nginx:
    image: nginx:latest
    ports:
      - 443:8443
    volumes:
      - ./nginx/nginx-setup.conf:/etc/nginx/conf.d/default.conf:ro
      - react_build:/var/www/client
    depends_on:
      - backend
      - frontend

  react_build:
